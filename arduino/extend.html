2   定时器及其中断

UNO中已有的3个定时器的用途：

  1. Timer0：8bit定时器，用于几个函数： delay() 、 millis() 、 micros()
  2. Timer1：16bit定时器，用于 Serve() 库
  3. Timer2：8bit定时器，用于 Tone() 库

而MEGA则有Timer3、4、5可用。

如果要使用的定时器与已有应用冲突，则已有应用的功能就不要再用了。

基本的LED闪耀，基于定时器中断:

#define LED 13
int timer1_counter;

void setup() {
    pinMode(LED, OUTPUT);
    noInterrupts();     //禁用所有中断
    TCCR1A=0;
    TCCR1B=0;
    timer1_counter=34286;   //中断周期=2Hz
    TCNT1=timer1_counter;
    TCCR1B |= (1<<CS12);    //256分频
    TIMSK  |= (1<<TOIE1);   //开启溢出中断
    interrupts();
}

ISR(TIMER1_OVF_vect) {
    TCNT1=timer1_counter;
    digitalWrite(LED, digitalRead(LED)^1);
}

void loop() {
    //...
}

由此可见，中断、寄存器设置之类的其实跟avr-gcc一样。

另一个使用CTC工作模式的例子:

#define LED 13

void setup() {
    pinMode(LED, OUTPUT);
    noInterrupts();
    TCCR1A=0;
    TCCR1B=0;
    TCNT1=0;
    OCR1A=31250;
    TCCR1B |= (1<<WGM12);   //CTC模式
    TCCR1B |= (1<<CS12);    //256分频
    TIMSK1 |= (1<<OCIE1A);  //比较匹配中断
    interrupts();
}

ISR(TIMER1_COMPA_vect) {
    digitalWrite(LED, digitalRead(LED) ^1);
}

void loop() {
    //...
}
